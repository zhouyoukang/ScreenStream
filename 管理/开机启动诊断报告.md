## 三、惑 — 解决方案

### 3.1 立即清理（笔记本 — 7个幽灵任务）

```powershell
# 删除笔记本幽灵计划任务
Unregister-ScheduledTask -TaskName 'WindsurfSetup_Zhou' -Confirm:$false
Unregister-ScheduledTask -TaskName 'WindsurfPack-SelfDestruct' -Confirm:$false
Unregister-ScheduledTask -TaskName 'GameAssistant' -Confirm:$false
Unregister-ScheduledTask -TaskName 'NodeBabyLinkRfile.exe' -Confirm:$false
Unregister-ScheduledTask -TaskName 'WindsurfDefenderShield' -Confirm:$false
Unregister-ScheduledTask -TaskName 'RDPWUpdater' -Confirm:$false

# FRP Client修复或删除（看是否还需要）
# Unregister-ScheduledTask -TaskName 'FRP Client' -Confirm:$false

# 清理幽灵用户的OneDrive任务
Get-ScheduledTask -TaskName 'OneDrive*' | Where-Object {
    $_.Principal.UserId -in @('32286','周','WsiAccount','AIOTVR')
} | Unregister-ScheduledTask -Confirm:$false

# 清理HP打印机任务
Get-ScheduledTask -TaskPath '\Hewlett-Packard\*','\HP\*' | Unregister-ScheduledTask -Confirm:$false
```
